338  subgraph = networkx.DiGraph()
339  if direction == "ascending":
340  todo = [(seed,n) for n in G.predecessors(seed)]
341  else:
342  todo = [(seed,n) for n in G.successors(seed)]

344  neighbor, current = todo.pop()
