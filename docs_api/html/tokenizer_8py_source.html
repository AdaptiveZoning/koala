70 def reverse_rpn(node, ast):
71  t = node.token
72  # print t.tvalue, t.ttype, t.tsubtype
73  if t.ttype == ExcelParserTokens.TOK_TYPE_FUNCTION: return t.tvalue + "(" + ",".join([reverse_rpn(x, ast) for x in node.children(ast)]) + ")"
74  elif t.ttype == ExcelParserTokens.TOK_TYPE_SUBEXPR: return "("
75  elif t.ttype == ExcelParserTokens.TOK_TYPE_SUBEXPR: return ")"
76  # TODO: add in RE substitution of " with "" for strings
77  elif t.ttype == ExcelParserTokens.TOK_TYPE_OPERAND: return t.tvalue
78  elif t.ttype == ExcelParserTokens.TOK_TYPE_OP_IN: return reverse_rpn(node.children(ast)[0], ast) + t.tvalue + reverse_rpn(node.children(ast)[1], ast)
79  elif t.ttype == ExcelParserTokens.TOK_TYPE_OP_PRE: return t.tvalue + reverse_rpn(node.children(ast)[0], ast)
